cmake_minimum_required(VERSION 3.18)
project(ShadowRealm VERSION 1.0.0 LANGUAGES CXX C)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

message(STATUS "ðŸŽ® Building ShadowRealm 3D")
message(STATUS "Platform: ${CMAKE_SYSTEM_NAME}")
message(STATUS "Generator: ${CMAKE_GENERATOR}")

# ============================================================================
#  DESKTOP BUILD  (Windows / macOS / Linux)
# ============================================================================
if(NOT ANDROID)
    message(STATUS "ðŸ–¥ï¸  Desktop build mode")

    find_package(OpenGL REQUIRED)
    message(STATUS "âœ… OpenGL found: ${OPENGL_LIBRARIES}")

    # ---- glad (OpenGL 3.3 core loader) -------------------------------------
    include(FetchContent)
    FetchContent_Declare(
        glad
        GIT_REPOSITORY https://github.com/Dav1dde/glad.git
        GIT_TAG        v2.0.8
        SOURCE_SUBDIR  cmake
    )
    FetchContent_MakeAvailable(glad)
    glad_add_library(glad_gl REPRODUCIBLE LOADER API gl:core=3.3)
    message(STATUS "âœ… GLAD downloaded and configured")

    # ---- GLFW --------------------------------------------------------------
    find_package(glfw3 3.3 QUIET)
    if(NOT glfw3_FOUND)
        message(WARNING "âš ï¸  GLFW3 not found via find_package â€“ trying find_library")
        find_library(GLFW_LIB glfw3)
        if(NOT GLFW_LIB)
            message(FATAL_ERROR "âŒ GLFW3 not found! Install with: apt-get install libglfw3-dev")
        endif()
    else()
        message(STATUS "âœ… GLFW3 found")
    endif()

    # ---- GLM ---------------------------------------------------------------
    if(NOT EXISTS "${CMAKE_SOURCE_DIR}/glm/glm/glm.hpp")
        message(FATAL_ERROR "âŒ GLM not found at glm/glm/glm.hpp")
    endif()
    message(STATUS "âœ… GLM found")

    # ---- ImGui -------------------------------------------------------------
    if(NOT EXISTS "${CMAKE_SOURCE_DIR}/imgui/imgui.h")
        message(FATAL_ERROR "âŒ ImGui not found at imgui/imgui.h")
    endif()
    if(NOT EXISTS "${CMAKE_SOURCE_DIR}/imgui/backends/imgui_impl_glfw.cpp")
        message(FATAL_ERROR "âŒ ImGui GLFW backend not found!")
    endif()
    if(NOT EXISTS "${CMAKE_SOURCE_DIR}/imgui/backends/imgui_impl_opengl3.cpp")
        message(FATAL_ERROR "âŒ ImGui OpenGL3 backend not found!")
    endif()
    message(STATUS "âœ… ImGui + backends found")

    # ---- Source file -------------------------------------------------------
    if(NOT EXISTS "${CMAKE_SOURCE_DIR}/main_desktop.cpp")
        message(FATAL_ERROR "âŒ main_desktop.cpp not found! (renamed from main.cpp)")
    endif()
    message(STATUS "âœ… Source file: main_desktop.cpp")

    set(IMGUI_SOURCES
        imgui/imgui.cpp
        imgui/imgui_draw.cpp
        imgui/imgui_tables.cpp
        imgui/imgui_widgets.cpp
        imgui/backends/imgui_impl_glfw.cpp
        imgui/backends/imgui_impl_opengl3.cpp
    )

    # ---- Target ------------------------------------------------------------
    add_executable(shadowrealm
        main_desktop.cpp
        ${IMGUI_SOURCES}
    )

    target_include_directories(shadowrealm PRIVATE
        ${CMAKE_SOURCE_DIR}/include
        ${CMAKE_SOURCE_DIR}/imgui
        ${CMAKE_SOURCE_DIR}/imgui/backends
        ${CMAKE_SOURCE_DIR}/glm
    )

    target_link_libraries(shadowrealm PRIVATE glad_gl)

    # OpenGL (platform differences)
    if(APPLE)
        target_link_libraries(shadowrealm PRIVATE ${OPENGL_LIBRARIES})
    else()
        if(TARGET OpenGL::OpenGL)
            target_link_libraries(shadowrealm PRIVATE OpenGL::OpenGL)
        else()
            target_link_libraries(shadowrealm PRIVATE ${OPENGL_LIBRARIES})
        endif()
    endif()

    # GLFW
    if(glfw3_FOUND)
        target_link_libraries(shadowrealm PRIVATE glfw)
    else()
        target_link_libraries(shadowrealm PRIVATE ${GLFW_LIB})
    endif()

    # Platform system libs
    if(APPLE)
        target_link_options(shadowrealm PRIVATE
            -framework Cocoa -framework IOKit -framework CoreVideo)
        message(STATUS "ðŸŽ macOS build configured")
    elseif(UNIX AND NOT APPLE)
        target_link_libraries(shadowrealm PRIVATE X11 dl pthread)
        message(STATUS "ðŸ§ Linux build configured")
    elseif(WIN32)
        message(STATUS "ðŸªŸ Windows build configured")
    endif()

    # Compile flags
    if(CMAKE_BUILD_TYPE STREQUAL "Release")
        if(MSVC)
            target_compile_options(shadowrealm PRIVATE /O2 /W4)
        else()
            target_compile_options(shadowrealm PRIVATE -O3 -march=native -flto)
        endif()
        message(STATUS "ðŸš€ Release build with optimizations")
    else()
        if(MSVC)
            target_compile_options(shadowrealm PRIVATE /Zi /W4)
        else()
            target_compile_options(shadowrealm PRIVATE -g -Wall -Wextra)
        endif()
        message(STATUS "ðŸ› Debug build")
    endif()

# ============================================================================
#  ANDROID BUILD  (arm64-v8a via NDK)
#  Active source:  main_android.cpp      (GLES 3.2 + EGL, full game loop)
#  Shelved source: main_android_vulkan.cpp (Vulkan 1.3, incomplete â€“ do not build)
# ============================================================================
else()
    message(STATUS "ðŸ¤– Android build mode")
    message(STATUS "   ABI:      ${ANDROID_ABI}")
    message(STATUS "   Platform: ${ANDROID_PLATFORM}")

    # ---- NDK glue ----------------------------------------------------------
    set(ANDROID_GLUE_DIR "${ANDROID_NDK}/sources/android/native_app_glue")
    if(NOT EXISTS "${ANDROID_GLUE_DIR}/android_native_app_glue.c")
        message(FATAL_ERROR "âŒ Android NDK native_app_glue not found at: ${ANDROID_GLUE_DIR}")
    endif()

    add_library(native_app_glue STATIC
        "${ANDROID_GLUE_DIR}/android_native_app_glue.c")
    target_include_directories(native_app_glue PUBLIC
        "${ANDROID_GLUE_DIR}")

    # ---- System libraries --------------------------------------------------
    find_library(LIB_GLES  GLESv3  REQUIRED)
    find_library(LIB_EGL   EGL     REQUIRED)
    find_library(LIB_LOG   log     REQUIRED)
    find_library(LIB_ANDROID android REQUIRED)

    # ---- Source file -------------------------------------------------------
    if(NOT EXISTS "${CMAKE_SOURCE_DIR}/main_android.cpp")
        message(FATAL_ERROR "âŒ main_android.cpp not found! (renamed from main.cpp / shadowrealm_mobile.cpp)")
    endif()
    message(STATUS "âœ… Source file: main_android.cpp (GLES 3.2)")

    # ---- Target (shared lib loaded by the APK) -----------------------------
    add_library(shadowrealm SHARED
        main_android.cpp)

    target_include_directories(shadowrealm PRIVATE
        ${CMAKE_SOURCE_DIR}/glm
        "${ANDROID_GLUE_DIR}")

    target_link_libraries(shadowrealm PRIVATE
        native_app_glue
        ${LIB_GLES}
        ${LIB_EGL}
        ${LIB_LOG}
        ${LIB_ANDROID})

    target_compile_options(shadowrealm PRIVATE
        -fPIC -fvisibility=hidden -Wall -Wextra)

    if(CMAKE_BUILD_TYPE STREQUAL "Release")
        target_compile_options(shadowrealm PRIVATE -O3 -flto)
        message(STATUS "ðŸš€ Release build with optimizations")
    else()
        target_compile_options(shadowrealm PRIVATE -g)
        message(STATUS "ðŸ› Debug build")
    endif()

    message(STATUS "âœ… Android (GLES) build configured")
    message(STATUS "   âš ï¸  Vulkan build (main_android_vulkan.cpp) is shelved â€“ not compiled")
endif()

message(STATUS "âœ… ShadowRealm 3D configured successfully")
